How to code basic stuff with SGDK

Basic setup in the main.c file

                    
                        #include "genesis.h"
                        #include "resources.h"

                        int main()
                        {
                            SPR_init();
                            while(1)
                            {
                                SPR_update();
                                SYS_doVBlankProcess();
                            }
                            return (0);
                        }

Show a text

VDP_drawText("Hello World", 11, 20);
  • 11 is the x position on screen
  • 20 is the y position on screen

You could also do like this :

                    
                        #include "genesis.h"
                        #include "resources.h"

                        s16 text_x_coord = 11;
                        s16 text_y_coord = 20;

                        int main()
                        {
                            VDP_drawText("Hello World", text_x_coord, text_y_coord);
                            SPR_init();
                            while(1)
                            {
                                SPR_update();
                                SYS_doVBlankProcess();
                            }
                            return (0);
                        }

So if you need to change its position dynamically, you can change the variable's value directly.

Remove a text

VDP_clearText(11, 20, 30);
  • 11 is the x position on screen
  • 20 is the y position on screen
  • 30 is the amount of characters to remove

You could also do like this :

                    
                        #include "genesis.h"
                        #include "resources.h"

                        // Text x coordinate
                        s16 text_x_coord = 11;
                
                        // Text y coordinate
                        s16 text_y_coord = 20;

                        // Number of characters
                        s16 text_nbr_char = 30;

                        int main()
                        {
                            // You see me
                            VDP_drawText("Hello World", text_x_coord, text_y_coord);

                            // You don't see me anymore
                            VDP_clearText(text_x_coord, text_y_coord, text_nbr_char);

                            SPR_init();

                            while(1)
                            {
                                SPR_update();
                                SYS_doVBlankProcess();
                            }
                            return (0);
                        }
